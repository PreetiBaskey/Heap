#include<bits/stdc++.h>
using namespace std;

vector<int> kThLargestElement(vector<int> &arr, int k) {
    priority_queue<int, vector<int>, greater<int>> minH;
    vector<int> result;
    for(int i = 0; i < arr.size(); i++) {
        minH.push(arr[i]);
        if(minH.size() > k) {
            minH.pop();
        }
    }
    
    while(!minH.empty()) {
        result.push_back(minH.top());
        minH.pop();
    }
    
    return result;
}

int main() {
    
    vector<int> arr = {7, 10, 4, 3, 20, 15};
    int k;
    cout<<"Enter value of K :- ";
    cin>>k;
    vector<int> result = kThLargestElement(arr, k);
    cout<<k<<"th largest elements are :- ";
    for(auto data: result) {
        cout<<data<<" ";
    }
    
    return 0;
}
