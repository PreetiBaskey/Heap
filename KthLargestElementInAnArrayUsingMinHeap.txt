#include<bits/stdc++.h>
using namespace std;

int main() {

    vector<int> arr = {3, 2, 1, 5, 6, 4};
    int k;
    priority_queue<int, vector<int>, greater<int>> mh;

    cout<<"Enter kth number to check :- ";
    cin>>k;
    for(int data: arr) {
        mh.push(data);

        if(mh.size() > k) {
            mh.pop();
        }
    }
    cout<<endl<<k<<"th largest element is :- "<<mh.top();
    //O(nlog(k))


    return 0;
}
