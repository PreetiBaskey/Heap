#include<bits/stdc++.h>
using namespace std;

vector<int> kThSmallestElement(vector<int> &arr, int k) {
    priority_queue<int> maxH;
    vector<int> result;
    for(int i = 0; i < arr.size(); i++) {
        maxH.push(arr[i]);
        if(maxH.size() > k) {
            maxH.pop();
        }
    }
    
    while(!maxH.empty()) {
        result.push_back(maxH.top());
        maxH.pop();
    }
    
    return result;
}

int main() {
    
    vector<int> arr = {7, 10, 4, 3, 20, 15};
    int k;
    cout<<"Enter value of K :- ";
    cin>>k;
    vector<int> result = kThSmallestElement(arr, k);
    cout<<k<<"th smallest elements are :- ";
    for(auto data: result) {
        cout<<data<<" ";
    }
    
    return 0;
}
