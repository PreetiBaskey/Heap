// time complexity - O(logn), space complexity - O(1)

#include<bits/stdc++.h>
using namespace std;

class Heap {
    public:
        int arr[100];
        int size;
        
        Heap() {
            arr[0] = -1;
            size = 0;
        }
        
        void insert(int data) {
            size = size + 1;
            int index = size;
            arr[index] = data;
            
            while(index > 1) {
                int parent = index/2;
                if(arr[parent] < arr[index]) {
                    swap(arr[parent], arr[index]);
                    index = parent;
                }
                else {
                    return;
                }
            }
        }
        
        void display() {
            for(int i = 1; i <= size; i++) {
                cout<<arr[i]<<" ";
            }
            cout<<endl;
        }
};

int main() {
    
    Heap h;
    h.insert(40);
    h.insert(30);
    h.insert(50);
    h.insert(10);
    h.insert(20);
    h.insert(100);
    h.display();
    
    return 0;
}
