// heapify function for 0-based indexing
#include<bits/stdc++.h>
using namespace std;

class Heap {
    public:
        int arr[100];
        int size;
        
        Heap() {
            size = 0;
        }
        
        void insert(int data) {
            arr[size] = data;
            int index = size;
            size++;
            
            while(index > 0) {
                int parent = (index - 1)/2;
                if(arr[parent] < arr[index]) {
                    swap(arr[parent], arr[index]);
                    index = parent;
                }
                else {
                    return;
                }
            }
        }
        
        void display() {
            for(int i = 0; i < size; i++) {
                cout<<arr[i]<<" ";
            }
            cout<<endl;
        }
        
        void deleteFromHeap() {
            if(size == 0) {
                cout<<"nothing to delete!";
                return;
            }
            arr[0] = arr[size - 1];
            size--;
            
            int index = 0;
            while(index < size) {
                int leftIndex = 2 * index + 1;
                int rightIndex = 2 * index + 2;
                int largest = index;
                
                if(leftIndex < size && arr[leftIndex] > arr[largest]) {
                    largest = leftIndex;
                }
                if(rightIndex < size && arr[rightIndex] > arr[largest]) {
                    largest = rightIndex;
                }
                if(largest != index) {
                    swap(arr[largest], arr[index]);
                    index = largest;
                }
                else {
                    return;                             
                }
            }
        }
};

void heapify(int arr[], int size, int index) {
    int leftIndex = 2 * index + 1;
    int rightIndex = 2 * index + 2;
    int largest = index;
    
    if(leftIndex < size && arr[leftIndex] > arr[largest]) {
        largest = leftIndex;
    }
    if(rightIndex < size && arr[rightIndex] > arr[largest]) {
        largest = rightIndex;
    }
    if(largest != index) {
        swap(arr[largest], arr[index]);
        heapify(arr, size, largest);
    }
}


int main() {
    
    Heap h;
    h.insert(40);
    h.insert(30);
    h.insert(50);
    h.insert(10);
    h.insert(20);
    h.insert(100);
    cout<<"heap after insertion of data :- ";
    h.display();
    
    h.deleteFromHeap();
    cout<<"heap after deletion of a data :- ";
    h.display();
    
    int arr[] = { 15, 5, 20, 1, 17, 10, 30 };
    int size = 7;
    for(int i = (size/2) - 1; i >= 0; i--) {
         heapify(arr, size, i);
    }
    cout<<"array after applying heapify algorithm :- ";
    for(int i = 0; i < size; i++) {
        cout<<arr[i]<<" ";
    }
    cout<<endl;
    
    return 0;
}



//heapify function for 1-based indexing
#include<bits/stdc++.h>
using namespace std;

void heapify(int arr[], int size, int i) {
    int largest = i;
    int leftIndex = 2 * i;
    int rightIndex = 2 * i + 1;
    
    if(leftIndex <= size && arr[largest] < arr[leftIndex]) {
        largest = leftIndex;
    }
    if(rightIndex <= size && arr[largest] < arr[rightIndex]) {
        largest = rightIndex;
    }
    if(largest != i) {
        int temp = arr[largest];
        arr[largest] = arr[i];
        arr[i] = temp;
        
        heapify(arr, size, largest);
    }
}

int main() {

    int arr[] = { -1, 15, 5, 20, 1, 17, 10, 30 };
    int size = 7;
    for(int i = size/2; i > 0; i--) {
        heapify(arr, size, i);
    }
    cout<<endl<<"Array after heapify :- ";
    for(int i = 1; i < size; i++) {
        cout<<arr[i]<<" ";
    }
    
    return 0;
}
